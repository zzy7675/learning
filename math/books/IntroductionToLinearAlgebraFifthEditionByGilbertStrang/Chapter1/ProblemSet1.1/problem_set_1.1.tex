\section*{Problem Set 1.1}

\subsection*{1.}

\begin{enumerate}[label=(\alph*)]
\item    
    \[
    \because
    \left[\begin{smallmatrix} 3 \\ 6 \\ 9\end{smallmatrix}\right] = 3 \left[\begin{smallmatrix} 1 \\ 2 \\ 3\end{smallmatrix}\right],\quad
    \left[\begin{smallmatrix} 1 \\ 2 \\ 3\end{smallmatrix}\right] = 1 \left[\begin{smallmatrix} 1 \\ 2 \\ 3\end{smallmatrix}\right] \quad
    \therefore 
    \left[\begin{smallmatrix} 3 \\ 6 \\ 9\end{smallmatrix}\right] \text{and} \left[\begin{smallmatrix} 1 \\ 2 \\ 3\end{smallmatrix}\right] \text{fills a line.}
    \]
\item 
    Let $v = \left[\begin{smallmatrix} 1 \\ 0 \\ 0\end{smallmatrix}\right], \quad w = \left[\begin{smallmatrix} 0 \\ 2 \\ 3\end{smallmatrix}\right]$, and we have all linear combinations $cv + dw$ where $c$ and $d$ are constants. This is the simplest form. Therefore $v$ and $w$ fill a plane.
\item 
    Let $u = \left[\begin{smallmatrix} 2 \\ 0 \\ 0\end{smallmatrix}\right], \quad v = \left[\begin{smallmatrix} 0 \\ 2 \\ 2\end{smallmatrix}\right], \quad w = \left[\begin{smallmatrix} 2 \\ 2 \\ 3\end{smallmatrix}\right].$
    $u$, $v$, $w$ are 3 different vectors, so $cu + dv + ew$ is the simplest form where $c,d,e$ are constants, and thus they fill all of $R^{3}$.
\end{enumerate}
